"""
파이프라인 오케스트레이터
재시도, 체크포인트, 로깅, 상태 관리 통합
"""
import json
import time
import shutil
from pathlib import Path
from typing import Any
from tenacity import retry, stop_after_attempt, wait_exponential, retry_if_exception_type
from config import CONFIG
from logging_config import setup_logger
from api_clients import get_clients
from tasks.script_gen import generate_script
from tasks.video_edit import edit_video

logger = setup_logger("pipeline")

class PipelineState:
    def __init__(self, state_file: Path = CONFIG.STATE_FILE):
        self.state_file = state_file
        self.state = {"last_stage": "init", "last_run_at": None, "last_trend_id": None, "last_video_id": None}
        self._load()

    def _load(self):
        if self.state_file.exists():
            with open(self.state_file, "r", encoding="utf-8") as f:
                self.state.update(json.load(f))

    def save(self):
        self.state_file.parent.mkdir(parents=True, exist_ok=True)
        with open(self.state_file, "w", encoding="utf-8") as f:
            json.dump(self.state, f, ensure_ascii=False, indent=2)
        logger.info(f"체크포인트 저장: {self.state_file}")

    def reset(self):
        self.state = {"last_stage": "init", "last_run_at": None, "last_trend_id": None, "last_video_id": None}
        self.save()

class ShortsPipeline:
    def __init__(self, state: PipelineState = None):
        self.state = state or PipelineState()
        self.clients = get_clients(mode="test")  # prod_mode="live" 시 실제 API 키 연동
        self.retry_kwargs = {
            "stop": stop_after_attempt(CONFIG.RETRY_MAX_ATTEMPTS),
            "wait": wait_exponential(multiplier=1, min=CONFIG.RETRY_BASE_DELAY_SEC, max=10),
            "retry": retry_if_exception_type((ConnectionError, TimeoutError, Exception)),
            "reraise": True
        }

    @retry(**retry_kwargs)
    def _run_stage(self, stage_name: str, func, *args, **kwargs):
        logger.info(f"🟡 [{stage_name}] 실행 시작")
        self.state["last_stage"] = stage_name
        self.state["last_run_at"] = time.strftime("%Y-%m-%d %H:%M:%S")
        self.state.save()
        result = func(*args, **kwargs)
        logger.info(f"🟢 [{stage_name}] 완료")
        return result

    def run(self):
        logger.info("🚀 쇼츠 자동 파이프라인 시작")
        try:
            # 1. 트렌드 데이터 수신
            trend = self._run_stage("trend_fetch", self.clients["trend"].fetch_trends)
            self.state["last_trend_id"] = trend.get("keyword", "unknown")
            
            # 2. 스크립트 생성 (0~3초 후크 강제)
            script = self._run_stage("script_gen", generate_script, trend)
            
            # 3. 영상 편집 (18초 고정, 하단 15% 자막 강제)
            # TODO: source_video는 실제 트래픽/소스에서 다운로드 또는 AI 생성 영상 경로
            source_video = CONFIG.WORK_DIR / "source_clip.mp4"
            source_video.parent.mkdir(parents=True, exist_ok=True)
            source_video.touch()  # 테스트용 빈 파일
            edited_video = self._run_stage("video_edit", edit_video, script, str(source_video))
            
            # 4. 자동 업로드 (월/수/금 07:30 연동)
            uploaded = self._run_stage("upload", self.clients["uploader"].upload_shorts, edited_video)
            self.state["last_video_id"] = uploaded.get("video_id", "unknown")
            self.state.save()
            
            logger.info(f"✅ 파이프라인 완료: {uploaded.get('url', 'N/A')}")
            return uploaded

        except Exception as e:
            logger.error(f"❌ 파이프라인 실패: {e}")
            self.state.save()
            raise

if __name__ == "__main__":
    state = PipelineState()
    pipeline = ShortsPipeline(state)
    pipeline.run()

쇼츠 자동화 파이프라인의 핵심 렌더링 엔진을 `ffmpeg-python` 기반으로 구현하고 테스트 환경을 구성합니다.

"""
ffmpeg_shorts.py
✅ 쇼츠 자동 렌더링 모듈
- 입력 영상을 18초로 자동 트리밍
- 하단 15% 안전지대에 자막 오버레이 (모바일 최적화)
- 실패 시 3회 재시도 로직 내장
"""
import ffmpeg
import time
import logging
import os
import sys
from typing import Optional

logging.basicConfig(
    level=logging.INFO,
    format='[%(asctime)s] %(levelname)s - %(message)s',
    datefmt='%H:%M:%S'
)
logger = logging.getLogger(__name__)

def trim_and_overlay_subtitles(
    input_path: str,
    output_path: str,
    font_path: str,
    subtitle_text: str,
    font_size: int = 24,
    font_color: str = 'white',
    border_w: int = 2,
    border_color: str = 'black@0.5',
    max_retries: int = 3,
    retry_delay: float = 2.0
) -> str:
    """
    영상을 18초로 단축하고 하단 15% 안전지대에 자막을 오버레이합니다.
    
    Args:
        input_path: 원본 영상 경로
        output_path: 출력 영상 경로
        font_path: 자막 폰트 파일 경로
        subtitle_text: 삽입할 자막 텍스트
        font_size: 폰트 크기
        font_color: 폰트 색상
        border_w: 자막 테두리 두께
        border_color: 자막 테두리 색상
        max_retries: 최대 재시도 횟수
        retry_delay: 재시도 간 대기 시간(초)
        
    Returns:
        str: 성공 시 출력 영상 경로
    """
    if not os.path.isfile(input_path):
        raise FileNotFoundError(f"입력 영상 파일을 찾을 수 없습니다: {input_path}")

    for attempt in range(1, max_retries + 1):
        try:
            logger.info(f"시도 {attempt}/{max_retries}: {os.path.basename(input_path)} 처리 중...")

            # 1. 입력 스트림 로드
            video = ffmpeg.input(input_path)

            # 2. 18초로 고정 트리밍 (시작 0초 기준)
            video_trimmed = video.trim(start=0, duration=18)
            video_trimmed = video_trimmed.set_fps(30)

            # 3. 자막 오버레이 (하단 15% 안전지대: y = h * 0.85)
            # iOS/Android 쇼츠 뷰어에서 자막이 잘리지 않도록 안전지대 적용
            video_with_subs = ffmpeg.drawtext(
                video_trimmed,
                text=subtitle_text,
                fontfile=font_path,
                fontsize=font_size,
                fontcolor=font_color,
                x='(w-text_w)/2',
                y='h*0.85',
                borderw=border_w,
                bordercolor=border_color
            )

            # 4. 오디오 처리 (오디오 트랙 존재 시 매핑)
            audio_in = video.audio
            if audio_in is not None:
                output_stream = ffmpeg.output(
                    video_with_subs, output_path,
                    vcodec='libx264', acodec='aac',
                    audio=audio_in,
                    preset='fast', crf=23,
                    movflags='+faststart'
                )
            else:
                output_stream = ffmpeg.output(
                    video_with_subs, output_path,
                    vcodec='libx264', preset='fast', crf=23,
                    movflags='+faststart'
                )

            # 5. 실행 및 검증
            output_stream.overwrite_output().run(capture_stdout=True, capture_stderr=True)

            logger.info("✅ 영상 처리 완료.")
            return output_path

        except ffmpeg.Error as e:
            stderr = e.stderr.decode('utf8') if e.stderr else "에러 상세 없음"
            logger.error(f"❌ ffmpeg 오류: {stderr}")
            if attempt < max_retries:
                logger.info(f"⏳ {retry_delay}초 후 재시도합니다...")
                time.sleep(retry_delay)
        except Exception as e:
            logger.error(f"❌ 시스템 예외: {e}")
            if attempt < max_retries:
                time.sleep(retry_delay)

    raise RuntimeError(f"최대 재시도 횟수({max_retries}) 초과로 실패했습니다.")


if __name__ == "__main__":
    # CLI 테스트 예시
    if len(sys.argv) < 5:
        print("Usage: python automation/ffmpeg_shorts.py <input.mp4> <output.mp4> <font.ttf> <subtitle_text>")
        sys.exit(1)

    try:
        result = trim_and_overlay_subtitles(
            input_path=sys.argv[1],
            output_path=sys.argv[2],
            font_path=sys.argv[3],
            subtitle_text=sys.argv[4]
        )
        print(f"SUCCESS: {result}")
    except Exception as e:
        print(f"FAILED: {e}")
        sys.exit(1)

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
📦 뽀득 쇼츠 자동화 파이프라인 스케줄러 (scheduler.py)
- 실행 시간: 월/수/금 07:30
- 동작: ffmpeg_shorts.py (편집) -> youtube_api_uploader.py (업로드)
- 특징: 3회 재시도 로직, 상세 로깅, croniter 기반 스케줄링
"""

import os
import sys
import time
import logging
import datetime
import subprocess
from croniter import croniter
from functools import wraps

# ------------------------------------------------------------------
# 설정 (CONFIG)
# ------------------------------------------------------------------
SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__))
LOG_DIR = os.path.join(SCRIPT_DIR, "logs")
os.makedirs(LOG_DIR, exist_ok=True)

# 로깅 설정
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s",
    handlers=[
        logging.FileHandler(os.path.join(LOG_DIR, f"scheduler_{datetime.date.today()}.log")),
        logging.StreamHandler(sys.stdout)
    ]
)
logger = logging.getLogger("PipelineScheduler")

# 스케줄 설정: 월(1), 수(3), 금(5) 07:30
CRON_EXPRESSION = "0 7 * * 1,3,5"
MAX_RETRIES = 3
RETRY_DELAY = 10  # 재시도 간격(초)

# ------------------------------------------------------------------
# 유틸리티 (UTILS)
# ------------------------------------------------------------------

def retry_on_failure(max_retries=MAX_RETRIES):
    """실패 시 재시도하는 데코레이터"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_retries + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    logger.error(f"[{func.__name__}] 실행 실패 (시도 {attempt}/{max_retries}): {e}")
                    if attempt == max_retries:
                        logger.critical(f"[{func.__name__}] 재시도 최대 횟수 도달. 파이프라인 중단.")
                        raise
                    logger.info(f"{RETRY_DELAY}초 후 재시도합니다...")
